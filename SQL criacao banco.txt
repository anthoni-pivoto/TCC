
CREATE SEQUENCE tb_lesoes_id_lesao_seq;

CREATE TABLE tb_lesoes (
                id_lesao BIGINT NOT NULL DEFAULT nextval('tb_lesoes_id_lesao_seq'),
                nm_lesao VARCHAR NOT NULL,
                CONSTRAINT tb_lesoes_pk PRIMARY KEY (id_lesao)
);


ALTER SEQUENCE tb_lesoes_id_lesao_seq OWNED BY tb_lesoes.id_lesao;

CREATE SEQUENCE tb_exercicios_id_exercicio_seq;

CREATE TABLE tb_exercicios (
                id_exercicio BIGINT NOT NULL DEFAULT nextval('tb_exercicios_id_exercicio_seq'),
                nm_exercicio VARCHAR NOT NULL,
                vid_path VARCHAR,
                CONSTRAINT tb_exercicios_pk PRIMARY KEY (id_exercicio)
);


ALTER SEQUENCE tb_exercicios_id_exercicio_seq OWNED BY tb_exercicios.id_exercicio;

CREATE TABLE tb_exercicio_lesao (
                id_lesao BIGINT NOT NULL,
                id_exercicio BIGINT NOT NULL,
                CONSTRAINT tb_exercicio_lesao_pk PRIMARY KEY (id_lesao, id_exercicio)
);


CREATE SEQUENCE tb_usuario_id_usuario_sequence;

CREATE TABLE tb_usuario (
                id_usuario BIGINT NOT NULL DEFAULT nextval('tb_usuario_id_usuario_seq'),
                nm_usuario VARCHAR NOT NULL,
                em_usuario VARCHAR NOT NULL,
                pwd_usuario VARCHAR NOT NULL,
                qtd_dias INTEGER,
                objetivo VARCHAR,
                peso REAL,
                altura REAL,
                CONSTRAINT id_usuario PRIMARY KEY (id_usuario)
);


ALTER SEQUENCE tb_usuario_id_usuario_seq OWNED BY tb_usuario.id_usuario;
